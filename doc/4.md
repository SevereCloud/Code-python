def heap(self, l):
    size = len(l)-1

    for i in range(size, -1, -1):
        if 2 * i + 1 <= size:
            if l[i] > l[2 * i + 1]:
                l[i], l[2 * i + 1] = l[2 * i + 1], l[i]
        if 2 * i + 2 <= size:
            if l[i] > l[2 * i + 2]:
                l[i], l[2 * i + 2] = l[2 * i + 2], l[i]
